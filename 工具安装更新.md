```

```

### conda常用命令

```python
# conda创建新环境
conda create -n name python=3.x
activate name 
conda deactivate
# 删除虚拟环境
conda remove -n env_name --all
# 删除包
conda remove -n env_name package_name
conda list # 查看安装了哪些包
conda env list 
conda info -e # 查看当前存在哪些虚拟环境
conda install -n env_name [package]
```
